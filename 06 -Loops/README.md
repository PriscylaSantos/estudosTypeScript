### Estruturas de repetição

#### while
[Exemplo](while.ts)
```typescript
while(condition)
{
    //executa enquanto a expressão seja verdadeira(true)
}
```

#### for
[Exemplo](for.ts)
```typescript
for((initial_count_value; termination-condition; step)
{
    //codigo
}
```

#### for...in
[Exemplo](forIn.ts)
```typescript
for (var val in list)
{
    //codigo
}
```

#### do...while
[Exemplo](doWhile.ts)
```typescript
do 
{
    //codigo

} while(condition)
```

#### break
[Exemplo](break.ts)
```typescript
while(condition)
{
    //codigo

    if (boolean_expression)
    {
        //codigo

        break
    }
}
```

#### continue
[Exemplo](continue.ts)
```typescript
for((initial_count_value; termination-condition; step)
{
    //codigo

    if (boolean_expression)
    {
        //codigo

        continue
    }
}
```

#### loop infinito
[Exemplo](infiniteLoop.ts)
```typescript
for(;;)
{
    //codigo
}
```
```typescript
while(true)
{
    //codigo
}
```
